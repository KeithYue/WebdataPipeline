    var o = getSrcElement(e);
    while(o && o.tagName != 'TR') o = o.parentNode;
    var overColor = '#E4EBF8';
    o.style.backgroundColor = overColor;
    var cells = o.cells;
    for(var i=0;i<cells.length;i++){
        cells[i].style.backgroundColor = overColor;
    }
}

function postListMouseOut(e){
    e = e || event;
    var o = e.fromElement;
    while(o && o.tagName != 'TR') o = o.parentNode;
    o.style.backgroundColor = o.srcBgColor;
    var cells = o.cells;
    for(var i=0;i<cells.length;i++){
        cells[i].style.backgroundColor = o.srcBgColor;
    }
}

function bindBgColorEvt(){
    var otbls = $tags('TABLE');
    var otbl = false;
    for(var i=0;i<otbls.length;i++){
        if(otbls[i].className == 'bbslist'){
            otbl = otbls[i];
            break;
        }
    }
    if(otbl){
        var trs = otbl.rows;
        for(var i=0;i<trs.length;i++){
            var tr = trs[i];
            tr.srcBgColor = tr.style.backgroundColor;
            addEvent(tr,'mouseover',postListMouseOver);
            addEvent(tr,'mouseout',postListMouseOut);
        }
    }
}
