    var o = e.fromElement;
    while(o && o.tagName != 'TR') o = o.parentNode;
    o.style.backgroundColor = o.srcBgColor;
    var cells = o.cells;
    for(var i=0;i<cells.length;i++){
        cells[i].style.backgroundColor = o.srcBgColor;
    }
}

function bindBgColorEvt(){

            otbl = otbls[i];
            break;
        }
    }
    if(otbl){

        for(var i=0;i<trs.length;i++){
            var tr = trs[i];
            tr.srcBgColor = tr.style.backgroundColor;

        }
    }
}
